    ```javascript
    // src/modules/login/domain/usecases/ValidateInstitutionalIdUseCase.js
    export class ValidateInstitutionalIdUseCase {
      constructor(authRepository) {
        this.authRepository = authRepository;
      }

      async execute(idInstitucional) {
        if (!idInstitucional) {
          throw new Error('ID institucional es requerido');
        }

        const user = await this.authRepository.validateInstitutionalId(
          idInstitucional
        );

        if (!user) {
          throw new Error('Usuario no encontrado');
        }

        return user;
      }
    }
    ```

  <Tab title="Infrastructure Layer">
    **External integrations** - API clients and concrete repository implementations.

    ```javascript
    // src/modules/login/infrastructure/api/apiClient.js
    import axios from 'axios';

    const apiClient = axios.create({
      baseURL: import.meta.env.VITE_API_URL || 'http://localhost:3000',
      timeout: 10000,
      headers: {
        'Content-Type': 'application/json',
      },
    });

    // Request interceptor
    apiClient.interceptors.request.use(
      (config) => {
        const token = localStorage.getItem('auth_token');
        if (token) {
          config.headers.Authorization = `Bearer ${token}`;
        }
        return config;
      },
      (error) => Promise.reject(error)
    );

    // Response interceptor
    apiClient.interceptors.response.use(
      (response) => response,
      (error) => {
        if (error.response?.status === 401) {
          localStorage.removeItem('auth_token');
          window.location.href = '/login';
        }
        return Promise.reject(error);
      }
    );

    export default apiClient;
    ```

    ```javascript
    // src/modules/login/infrastructure/repositories/AuthRepositoryImpl.js
    import { AuthRepository } from '../../domain/repositories/AuthRepository';
    import apiClient from '../api/apiClient';
    import { User } from '../../domain/entities/User';

    export class AuthRepositoryImpl extends AuthRepository {
      async validateInstitutionalId(idInstitucional) {
        try {
          const response = await apiClient.post('/api/auth/login', {
            id_institucional: idInstitucional,
          });

          const userData = response.data.usuario;

          return new User({
            id: userData.id,
            idInstitucional: userData.id_institucional,
            documentoIdentidad: userData.documento_identidad,
            nombreCompleto: userData.nombre_completo,
            correoInstitucional: userData.correo_institucional,
            estadoGeneral: userData.estado_general,
            totalFallas: userData.total_fallas,
            roles: userData.roles,
          });
        } catch (error) {
          throw new Error(
            error.response?.data?.message || 'Error al validar ID'
          );
        }
      }
    }
    ```

    **React hooks** - Orchestrate use cases and manage UI state.

    ```javascript
    // src/modules/login/application/hooks/useLogin.js
    import { useState } from 'react';
    import { useNavigate } from 'react-router-dom';
    import { ValidateInstitutionalIdUseCase } from '../../domain/usecases/ValidateInstitutionalIdUseCase';
    import { AuthRepositoryImpl } from '../../infrastructure/repositories/AuthRepositoryImpl';

    export const useLogin = () => {
      const [loading, setLoading] = useState(false);
      const [error, setError] = useState(null);
      const navigate = useNavigate();

      // Dependency injection
      const authRepository = new AuthRepositoryImpl();
      const validateIdUseCase = new ValidateInstitutionalIdUseCase(
        authRepository
      );

      const login = async (idInstitucional) => {
        try {
          setLoading(true);
          setError(null);

          // Execute use case
          const user = await validateIdUseCase.execute(idInstitucional);

          // Navigate to profile
          navigate('/user-profile', {
            state: { user }
          });

          return user;
        } catch (err) {
          setError(err.message);
          throw err;
        } finally {
          setLoading(false);
        }
      };

      return { login, loading, error };
    };
    ```

### Routing Architecture

```jsx
// src/App.jsx
import React from 'react';
import { BrowserRouter as Router, Routes, Route, Navigate } from 'react-router-dom';
import { LoginPage }       from './modules/login/presentation/pages/LoginPage';
import { UserProfilePage } from './modules/user-profile/presentation/pages/UserProfilePage';
import { AdminPage }       from './modules/admin/presentation/pages/AdminPage';
import { ProtectedRoute }  from './shared/components/ProtectedRoute';

function App() {
  return (
    <Router>
      <Routes>
        {/* Public route */}
        <Route path="/login" element={<LoginPage />} />

        {/* Protected route - requires authentication */}
        <Route
          path="/user-profile"
          element={
            <ProtectedRoute>
              <UserProfilePage />
            </ProtectedRoute>
          }
        />

        {/* Admin panel */}
        <Route path="/admin" element={<AdminPage />} />

        {/* Default redirect */}
        <Route path="/" element={<Navigate to="/login" replace />} />
      </Routes>
    </Router>
  );
}

export default App;
```

### Core Tables

<Tabs>
  <Tab title="Usuarios (Users)">
    Main user table storing basic information.

    ```sql
    CREATE TABLE Usuarios (
      id INTEGER PRIMARY KEY AUTO_INCREMENT,
      id_institucional VARCHAR(20) UNIQUE NOT NULL,
      documento_identidad VARCHAR(20) UNIQUE NOT NULL,
      nombre_completo VARCHAR(255) NOT NULL,
      correo_institucional VARCHAR(255) UNIQUE NOT NULL,
      estado_general ENUM('activo', 'bloqueado') DEFAULT 'activo',
      total_fallas INTEGER DEFAULT 0,
      created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
      updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
      
      INDEX idx_id_institucional (id_institucional),
      INDEX idx_estado (estado_general),
      INDEX idx_fallas (total_fallas)
    );
    ```

    **Key Fields:**
    - `id_institucional`: Unique ID assigned by UCC (e.g., 80123456)
    - `estado_general`: User status ('activo' or 'bloqueado')
    - `total_fallas`: Accumulated failure counter
  </Tab>

  <Tab title="Roles">
    Defines available roles in the system.

    ```sql
    CREATE TABLE Roles (
      id INTEGER PRIMARY KEY AUTO_INCREMENT,
      nombre_rol VARCHAR(50) NOT NULL,
      descripcion TEXT,
      created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    );

    -- Initial data
    INSERT INTO Roles (id, nombre_rol, descripcion) VALUES
      (1, 'Estudiante', 'Usuario estudiante de la universidad'),
      (2, 'Empleado', 'Empleado de la universidad'),
      (3, 'Contratista', 'Contratista externo');
    ```

    **Available Roles:**
    1. Estudiante (Student)
    2. Empleado (Employee)
    3. Contratista (Contractor)
  </Tab>

  <Tab title="Usuario_Roles">
    Many-to-many relationship between users and roles.

    ```sql
    CREATE TABLE Usuario_Roles (
      usuario_id INTEGER NOT NULL,
      rol_id INTEGER NOT NULL,
      fecha_asignacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
      activo BOOLEAN DEFAULT TRUE,
      
      PRIMARY KEY (usuario_id, rol_id),
      FOREIGN KEY (usuario_id) REFERENCES Usuarios(id) ON DELETE CASCADE,
      FOREIGN KEY (rol_id) REFERENCES Roles(id)
    );
    ```

    **Example:** A user can be both Student AND Employee:
    ```sql
    INSERT INTO Usuario_Roles (usuario_id, rol_id) VALUES
      (1, 1), -- User 1 is a Student
      (1, 2); -- User 1 is also an Employee
    ```
  </Tab>

  <Tab title="Registro_Fallas">
    Historical record of all failures.

    ```sql
    CREATE TABLE Registro_Fallas (
      id INTEGER PRIMARY KEY AUTO_INCREMENT,
      usuario_id INTEGER NOT NULL,
      fecha_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
      motivo ENUM('olvido', 'perdida', 'robo') NOT NULL,
      observacion TEXT,
      sede_registro VARCHAR(100),
      vigilante_registro VARCHAR(255),
      
      FOREIGN KEY (usuario_id) REFERENCES Usuarios(id),
      INDEX idx_usuario_fecha (usuario_id, fecha_hora),
      INDEX idx_motivo (motivo)
    );
    ```

    **Motivo Options:**
    - `olvido`: User forgot the card
    - `perdida`: User lost the card
    - `robo`: Card was stolen
  </Tab>

  <Tab title="Historial_Bloqueos">
    History of blocks and unblocks.

    ```sql
    CREATE TABLE Historial_Bloqueos (
      id INTEGER PRIMARY KEY AUTO_INCREMENT,
      usuario_id INTEGER NOT NULL,
      fecha_bloqueo TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
      motivo_bloqueo ENUM(
        'limite_fallas',
        'perdida_carnet',
        'robo_carnet',
        'administrativo'
      ) NOT NULL,
      observacion_bloqueo TEXT,
      fecha_desbloqueo TIMESTAMP NULL,
      desbloqueado_por INTEGER NULL,
      observacion_desbloqueo TEXT NULL,
      estado ENUM('activo', 'resuelto') DEFAULT 'activo',
      
      FOREIGN KEY (usuario_id) REFERENCES Usuarios(id),
      FOREIGN KEY (desbloqueado_por) REFERENCES Administradores(id),
      INDEX idx_usuario (usuario_id),
      INDEX idx_estado (estado)
    );
    ```

    **Block Reasons:**
    - `limite_fallas`: 4th "olvido" triggered automatic block
    - `perdida_carnet`: User reported card lost
    - `robo_carnet`: User reported card stolen
    - `administrativo`: Administrative block by staff
  </Tab>
</Tabs>

### Business Logic Queries

<CodeGroup>
```sql Check User Status
-- Get user status and failure count
SELECT 
  u.id,
  u.id_institucional,
  u.nombre_completo,
  u.estado_general,
  u.total_fallas,
  COUNT(rf.id) as total_registros_fallas
FROM Usuarios u
LEFT JOIN Registro_Fallas rf ON u.id = rf.usuario_id
WHERE u.id_institucional = '80123456'
GROUP BY u.id;
```

```sql Register Olvido (Forgot)
-- Register "forgot" failure and check if user should be blocked
START TRANSACTION;

-- Insert failure record
INSERT INTO Registro_Fallas (
  usuario_id,
  motivo,
  sede_registro
) VALUES (
  1,
  'olvido',
  'Campus Bogotá'
);

-- Increment counter
UPDATE Usuarios
SET total_fallas = total_fallas + 1
WHERE id = 1;

-- Check if should block (4+ failures)
UPDATE Usuarios
SET estado_general = 'bloqueado'
WHERE id = 1 AND total_fallas >= 4;

-- If blocked, create block record
INSERT INTO Historial_Bloqueos (
  usuario_id,
  motivo_bloqueo,
  observacion_bloqueo
)
SELECT 
  id,
  'limite_fallas',
  'Bloqueado automáticamente por 4 olvidos'
FROM Usuarios
WHERE id = 1 AND estado_general = 'bloqueado';

COMMIT;
```

```sql Register Loss/Theft
-- Immediate block for lost or stolen card
START TRANSACTION;

-- Insert failure record (without incrementing counter)
INSERT INTO Registro_Fallas (
  usuario_id,
  motivo,
  sede_registro
) VALUES (
  1,
  'perdida',  -- or 'robo'
  'Campus Bogotá'
);

-- Block user immediately
UPDATE Usuarios
SET estado_general = 'bloqueado'
WHERE id = 1;

-- Create block record
INSERT INTO Historial_Bloqueos (
  usuario_id,
  motivo_bloqueo,
  observacion_bloqueo
) VALUES (
  1,
  'perdida_carnet',  -- or 'robo_carnet'
  'Usuario reportó pérdida de carnet'
);

COMMIT;
```

```sql Unblock User
-- Admin unblocks a user
START TRANSACTION;

-- Activate user
UPDATE Usuarios
SET 
  estado_general = 'activo',
  total_fallas = 0  -- Reset counter
WHERE id = 1;

-- Update block record
UPDATE Historial_Bloqueos
SET 
  fecha_desbloqueo = NOW(),
  desbloqueado_por = 5,  -- Admin ID
  observacion_desbloqueo = 'User brought new card',
  estado = 'resuelto'
WHERE usuario_id = 1 AND estado = 'activo';

COMMIT;
```

```sql Dashboard Stats
-- Get statistics for admin dashboard
SELECT
  -- Blocked users
  SUM(CASE WHEN estado_general = 'bloqueado' THEN 1 ELSE 0 END) as bloqueados,
  
  -- High risk (3 failures)
  SUM(CASE WHEN total_fallas = 3 THEN 1 ELSE 0 END) as riesgo_alto,
  
  -- Medium risk (2 failures)
  SUM(CASE WHEN total_fallas = 2 THEN 1 ELSE 0 END) as con_2_fallas,
  
  -- Low risk (1 failure)
  SUM(CASE WHEN total_fallas = 1 THEN 1 ELSE 0 END) as con_1_falla,
  
  -- No failures
  SUM(CASE WHEN total_fallas = 0 THEN 1 ELSE 0 END) as sin_fallas
FROM Usuarios;
```
